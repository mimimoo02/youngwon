베스트 모델 선정

96개 결과 중 회귀=R², 분류=F1_weighted 기준으로 센터×태스크별 최적 모델(총 8개)을 선정.

for center in centers:
            for task in ['regression', 'classification']:
                center_task_data = results_df[(results_df['center'] == center) & (results_df['task'] == task)]
                if len(center_task_data) == 0:
                    continue
                if task == 'regression':
                    best_model = center_task_data.loc[center_task_data['R2'].idxmax()]
                    print(f"{center} 회귀: {best_model['model']} ({best_model['split_method']}) - R2: {best_model['R2']:.4f}")
                else:
                    best_model = center_task_data.loc[center_task_data['F1_weighted'].idxmax()]
                    print(f"{center} 분류: {best_model['model']} ({best_model['split_method']}) - F1: {best_model['F1_weighted']:.4f}")
                best_models.append(best_model.to_dict())
